define('modules/pdp/controllers/AddToBasketController', [
  'domlib',
  'modules/mvc/fn',
  'modules/pdp/controllers/FormHandlerController',
  'modules/pdp/views/AddToBasketView',
  'modules/tesco.analytics',
  'modules/chip-and-pin/kmf-io',
  'modules/google-analytics/pdpTracking',
  'modules/recommenders/common'
], function ($,
  fn,
  FormHandlerController,
  AddToBasketView,
  analytics,
  kmfIO,
  googleEC,
  recommenders
) {
  'use strict';
 
  /**
   *
   * @param {Object} model The add to basket view.
   * @return {void}
   */
  function AddToBasketController(model) {
    this.sTag = 'addToBasket';
    this.oTooltipStyles = {
      kiosk: ['tooltip-left'],
      allDesktops: ['tooltip-left'],
      allDevices: ['tooltip-bottom']
    };
    this.views = {
      classes: {
        AddToBasketView: AddToBasketView
      }
    };
    this.parent.constructor.call(this, model);
  }
 
  fn.inherit(AddToBasketController, FormHandlerController);
 
  AddToBasketController.prototype._initDependancies = function () {
    this.customerOriginRichRelevanceData = recommenders.richRelevanceOriginAnalyticsHandler();
  };
 
  AddToBasketController.prototype._bindViewEvents = function (oData) {
    $(oData.oView.oElms.elWrapper).on(
      'click', 'button', { oView: oData.oView },
      this._onAddToBasket.bind(this)
    );
 
    if (oData.oView.oElms.elQtyInput) {
      $(oData.oView.oElms.elWrapper)
        .on(
          'change', 'input.product-quantity',
          { oView: oData.oView },
          this._onQuantityChange.bind(this)
        )
        .on(
          'keyup', 'input.product-quantity',
          { oView: oData.oView },
          this._onQuantityKeyup.bind(this)
        )
        .on(
          'keypress', 'input.product-quantity',
          { oView: oData.oView },
          this._handleQuantityKeypress.bind(this)
        );
    }
  };
 
  AddToBasketController.prototype._onAddToBasket = function (oEvent) {
    var oView = oEvent.data.oView;
 
    $(oView.oElms.elSubmitBtn).addClass('submitting');
 
    this._activeView = oView;
    this.oModel.send({
      oPromise: {
        sSearchKey: $(oView.oElms.elSubmitBtn).data('mvc-key'),
        mSearchValue: $(oView.oElms.elSubmitBtn).data('mvc-value')
      },
      fnDoneCallback: this._onAddToBasketSuccess.bind(this),
      fnFailCallback: this._onAddToBasketFailure.bind(this)
    });
  };
 
  AddToBasketController.prototype._onAddToBasketSuccess = function (mRespData) {
    var _this = this,
      viewData = this._activeView.oData.mvc,
      viewElms = this._activeView.oElms,
      quanityAdded = 0,
      basketQuantity = 0;
 
    if (mRespData.hasOwnProperty('success')) {
      fn.createEvent({
        name: 'addToBasketSuccess',
        namespace: this._activeView.id,
        propName: 'oData',
        data: mRespData
      }).fire();
 
      quanityAdded = mRespData.success.data.itemsAdded.quantity;
      basketQuantity = mRespData.success.data.basket.total;
 
      kmfIO.enableBasketButton(basketQuantity);
 
      if (mRespData.success.data) {
        this.setEvent({
          sName: 'dataFetched',
          mfetchedData: mRespData.success.data
        }, false, true);
      }
 
      googleEC.setAnalytics(mRespData.success.data.itemsAdded.items, 'addToBasket');
      this._analyticsTracking(mRespData.success.analytics);
 
      this.renderOverlay({
        sTag: 'basketOverlay',
        sClassNames: 'basket-overlay with-header',
        oStyles: {
          allDesktops: ['lightbox', 'fullScreen'],
          allDevices: ['slideIn-left', 'fullScreen']
        },
        sTitle: quanityAdded > 1
            ? '<strong>' + quanityAdded + ' items</strong> added to basket'
            : '<strong>' + quanityAdded + ' item</strong> added to basket',
        sOverlayContent: mRespData.success.html,
        toDestroyOnClose: true,
        forceUpdateActive: true,
        callback: function overlayRenderedCallback(overlaySelector) {
          var skuData = viewData.sku,
            event = null;
 
          if (_this.sanityCheckData(skuData).objects) {
            if (_this.isObject(skuData.personalisation, true)
                && skuData.personalisation.isPersonalised) {
              event = new $.Event('resetPersonalisation');
              event.oData = { skuId: skuData.id };
              $(window).trigger(event);
            }
          }
 
          _this.addOnMsgTooltip(overlaySelector);
        }
      });
    } else {
      this.setEvent({
        sName: 'addToBasketFailure',
        oRespData: mRespData
      }, false, true);
 
      this.createTooltip({
        sTooltopMessage: mRespData.failure.message,
        elTrigger: viewElms.elQtyInput || viewElms.elSubmitBtn,
        sType: 'error'
      });
    }
 
    $(viewElms.elSubmitBtn).removeClass('submitting');
  };
 
  AddToBasketController.prototype._onAddToBasketFailure = function (sErrorMsg, oJqXHR) {
    this.setEvent({
      sName: 'addToBasketFailure',
      oRespData: oJqXHR
    }, false, true);
 
    this.createTooltip({
      sTooltopMessage: sErrorMsg,
      elTrigger: this._activeView.oElms.elQtyInput || this._activeView.oElms.elSubmitBtn,
      sType: 'error'
    });
 
    $(this._activeView.oElms.elSubmitBtn).removeClass('submitting');
  };
 
  AddToBasketController.prototype._onQuantityChange = function (oEvent) {
    var oView = oEvent.data.oView;
 
    this.oModel.update({
      sSearchKey: $(oView.oElms.elSubmitBtn).data('mvc-key'),
      mSearchValue: $(oView.oElms.elSubmitBtn).data('mvc-value'),
      sUpdateKey: 'oData',
      mUpdateValue: null,
      sUpdatePropKey: $(oView.oElms.elQtyInput).data('mvc-key'),
      mUpdatePropValue: oView.oElms.elQtyInput.value
    });
  };
 
  AddToBasketController.prototype._onQuantityKeyup = function (oEvent) {
    var MESSAGE = 'Please enter a valid number.',
      oView = oEvent.data.oView,
      oRegex = /^([0-9]+)$|^$/,
      sQtyValue = oView.oElms.elQtyInput.value;
 
    if (!sQtyValue.match(oRegex)) {
      this.createTooltip({
        sTooltopMessage: MESSAGE,
        elTrigger: oView.oElms.elQtyInput,
        sType: 'error'
      });
    } else if (this._activePanel.tooltip) {
      this._activePanel.tooltip.close();
    }
  };
 
  AddToBasketController.prototype._handleQuantityKeypress = function (event) {
    if (event.which === 13) {
      event.preventDefault();
      this._onQuantityChange(event);
      this._onAddToBasket(event);
      $(event.currentTarget).blur();
    }
  };
 
  AddToBasketController.prototype._analyticsTracking = function (analyticsResponse) {
    var _oWebAnalytics = null,
      extResp = null,
      v = [],
      sRelAnalyticsText = null,
      extendedVars = null;
 
    if (this._activeView.oData.mvc.info.relationship !== undefined) {
      if (this._activeView.oData.mvc.info.relationship === 'completeTheLook') {
        sRelAnalyticsText = 'Complete the look';
      }
      if (this._activeView.oData.mvc.info.relationship === 'outFit') {
        sRelAnalyticsText = 'Outfit Block';
      }
    } else {
      sRelAnalyticsText = 'sidebar';
    }
 
    _oWebAnalytics = new analytics.WebMetrics();
    extResp = analyticsResponse[0].events + ',event32,event45';
    v = [{
      eVar45: 'add to basket',
      prop19: 'add to basket',
      eVar59: 'pdp - ' + sRelAnalyticsText + ' - add to basket',
      prop42: 'pdp - ' + sRelAnalyticsText + ' - add to basket',
      events: extResp
    }];
 
    if (this.customerOriginRichRelevanceData !== null) {
      v[0].linkTrackVars = 'prop39,eVar49,eVar61';
      v[0].list3 = this.customerOriginRichRelevanceData;
      v[0].prop39 = 'rich releance pdp clickthrough';
      v[0].eVar49 = 'prop39';
      v[0].eVar61 = 'PDP - rich relevance';
    }
 
    extendedVars = $.extend(true, [{}], analyticsResponse, v);
    _oWebAnalytics.submit(extendedVars);
  };
 
  AddToBasketController.prototype.addOnMsgTooltip = function (overlaySelector) {
    var $overlay = $(overlaySelector),
      $buttonActionsContainer = $overlay.find('.basket-overlay__actions'),
      $checkoutButton = $buttonActionsContainer.find('#basket-checkout-mini');
 
    if ($checkoutButton.hasClass('disabled')) {
      $checkoutButton.on('click', this._openAddOnMessageTooltip.bind(this));
    }
  };
 
  AddToBasketController.prototype._openAddOnMessageTooltip = function (event) {
    var $elCheckBtn = $(event.currentTarget),
      $message = $('#tooltipMsg').val();
 
    this.createTooltip({
      sTooltopMessage: $message,
      elTrigger: $elCheckBtn
    });
  };
  return AddToBasketController;
});
